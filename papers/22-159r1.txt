[10:Examples]
Change
   CALL SPLIT (STRING, TOKENS, SET)
  
to
   CALL TOKENIZE (STRING, SET, TOKENS)

[17:Examples]

   CALL SPLIT (STRING, SET, FIRST, LAST)
